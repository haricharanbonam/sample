```dart
import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(
  home: CricketApp(),
  debugShowCheckedModeBanner: false,
  theme: ThemeData(primarySwatch: Colors.green),
));

class CricketApp extends StatefulWidget {
  @override
  _CricketAppState createState() => _CricketAppState();
}

class _CricketAppState extends State<CricketApp> {
  int score = 0, wickets = 0;
  double overs = 0.0;
  List<Map<String, dynamic>> records = [];
  
  final scoreCtrl = TextEditingController();
  final wicketsCtrl = TextEditingController();
  final oversCtrl = TextEditingController();

  void recordScore() {
    setState(() {
      score = int.tryParse(scoreCtrl.text) ?? 0;
      wickets = int.tryParse(wicketsCtrl.text) ?? 0;
      overs = double.tryParse(oversCtrl.text) ?? 0.0;
      records.add({'score': score, 'wickets': wickets, 'overs': overs});
      scoreCtrl.clear();
      wicketsCtrl.clear();
      oversCtrl.clear();
    });
  }

  Widget inputField(String label, IconData icon, TextEditingController ctrl) {
    return TextField(
      controller: ctrl,
      keyboardType: TextInputType.numberWithOptions(decimal: true),
      decoration: InputDecoration(
        labelText: label,
        border: OutlineInputBorder(),
        prefixIcon: Icon(icon),
      ),
    );
  }

  Widget statBox(String label, String value, Color color) {
    return Column(
      children: [
        Text(label, style: TextStyle(fontSize: 16, color: Colors.grey[700])),
        Text(value, style: TextStyle(fontSize: 32, fontWeight: FontWeight.bold, color: color)),
      ],
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Cricket Score Recorder'), centerTitle: true),
      body: SingleChildScrollView(
        padding: EdgeInsets.all(20),
        child: Column(
          children: [
            Text('Record Cricket Match', style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold, color: Colors.green[800])),
            SizedBox(height: 20),
            inputField('Score', Icons.sports_cricket, scoreCtrl),
            SizedBox(height: 15),
            inputField('Wickets', Icons.close, wicketsCtrl),
            SizedBox(height: 15),
            inputField('Overs', Icons.timelapse, oversCtrl),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: recordScore,
              child: Text('Record Score', style: TextStyle(fontSize: 18)),
              style: ElevatedButton.styleFrom(
                backgroundColor: Colors.green,
                foregroundColor: Colors.white,
                padding: EdgeInsets.all(15),
                minimumSize: Size(double.infinity, 50),
              ),
            ),
            SizedBox(height: 10),
            OutlinedButton(
              onPressed: () => setState(() => records.clear()),
              child: Text('Clear All'),
              style: OutlinedButton.styleFrom(
                padding: EdgeInsets.all(15),
                minimumSize: Size(double.infinity, 50),
              ),
            ),
            SizedBox(height: 30),
            Text('Summary Display', style: TextStyle(fontSize: 22, fontWeight: FontWeight.bold, color: Colors.green[800])),
            SizedBox(height: 15),
            if (records.isEmpty)
              Container(
                padding: EdgeInsets.all(20),
                decoration: BoxDecoration(
                  color: Colors.grey[200],
                  borderRadius: BorderRadius.circular(10),
                ),
                child: Text('No records yet', textAlign: TextAlign.center, style: TextStyle(fontSize: 16)),
              )
            else
              Column(
                children: [
                  Container(
                    padding: EdgeInsets.all(20),
                    decoration: BoxDecoration(
                      color: Colors.green[50],
                      borderRadius: BorderRadius.circular(10),
                      border: Border.all(color: Colors.green),
                    ),
                    child: Column(
                      children: [
                        Text('Current Match Status', style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold)),
                        SizedBox(height: 15),
                        Row(
                          mainAxisAlignment: MainAxisAlignment.spaceAround,
                          children: [
                            statBox('Score', '$score', Colors.green[800]!),
                            statBox('Wickets', '$wickets', Colors.red[700]!),
                            statBox('Overs', '$overs', Colors.blue[700]!),
                          ],
                        ),
                      ],
                    ),
                  ),
                  SizedBox(height: 20),
                  Text('All Records', style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold)),
                  SizedBox(height: 10),
                  ...records.asMap().entries.map((entry) {
                    int idx = entry.key;
                    var rec = entry.value;
                    return Card(
                      margin: EdgeInsets.symmetric(vertical: 5),
                      child: ListTile(
                        leading: CircleAvatar(
                          backgroundColor: Colors.green,
                          child: Text('${idx + 1}', style: TextStyle(color: Colors.white)),
                        ),
                        title: Text('Score: ${rec['score']} | Wickets: ${rec['wickets']} | Overs: ${rec['overs']}'),
                      ),
                    );
                  }).toList(),
                ],
              ),
          ],
        ),
      ),
    );
  }
}
```

```text
MaterialApp
└── CricketApp (Scaffold)
    ├── AppBar (Title: 'Cricket Score Recorder')
    └── Body: SingleChildScrollView
        └── Column
            ├── Text ('Record Cricket Match')
            ├── TextField (Score - with Icon)
            ├── TextField (Wickets - with Icon)
            ├── TextField (Overs - with Icon)
            ├── ElevatedButton ('Record Score')
            ├── OutlinedButton ('Clear All')
            ├── Text ('Summary Display')
            └── Conditional Display:
                ├── If Empty: Container (Grey box - 'No records yet')
                └── If Has Records: Column
                    ├── Container (Green box - Current Match Status)
                    │   └── Row
                    │       ├── Column (Score)
                    │       ├── Column (Wickets)
                    │       └── Column (Overs)
                    ├── Text ('All Records')
                    └── List of Cards
                        └── Card (Each record)
                            └── ListTile
                                ├── CircleAvatar (Number)
                                └── Text (Score|Wickets|Overs)
```
