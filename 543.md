import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(
  home: BirthdayApp(),
  debugShowCheckedModeBanner: false,
  theme: ThemeData(primarySwatch: Colors.pink),
));

class BirthdayApp extends StatefulWidget {
  @override
  _BirthdayAppState createState() => _BirthdayAppState();
}

class _BirthdayAppState extends State<BirthdayApp> with SingleTickerProviderStateMixin {
  List<Map<String, String>> birthdays = [];
  final nameCtrl = TextEditingController();
  final dateCtrl = TextEditingController();
  
  late AnimationController animCtrl;
  late Animation<double> scaleAnim;
  bool showWish = false;
  String wishName = '';

  @override
  void initState() {
    super.initState();
    animCtrl = AnimationController(duration: Duration(milliseconds: 600), vsync: this);
    scaleAnim = Tween<double>(begin: 0.0, end: 1.0).animate(
      CurvedAnimation(parent: animCtrl, curve: Curves.bounceOut),
    );
  }

  @override
  void dispose() {
    animCtrl.dispose();
    nameCtrl.dispose();
    dateCtrl.dispose();
    super.dispose();
  }

  void addBirthday() {
    if (nameCtrl.text.isNotEmpty && dateCtrl.text.isNotEmpty) {
      setState(() {
        birthdays.add({'name': nameCtrl.text, 'date': dateCtrl.text});
        nameCtrl.clear();
        dateCtrl.clear();
      });
    }
  }

  void showBirthdayWish(String name) {
    setState(() {
      wishName = name;
      showWish = true;
    });
    animCtrl.reset();
    animCtrl.forward();
    Future.delayed(Duration(seconds: 2), () {
      setState(() => showWish = false);
    });
  }

  void deleteBirthday(int index) {
    setState(() => birthdays.removeAt(index));
  }

  Widget inputField(String label, IconData icon, TextEditingController ctrl) {
    return TextField(
      controller: ctrl,
      decoration: InputDecoration(
        labelText: label,
        border: OutlineInputBorder(),
        prefixIcon: Icon(icon),
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Birthday Reminder'), centerTitle: true),
      body: Stack(
        children: [
          Padding(
            padding: EdgeInsets.all(20),
            child: Column(
              children: [
                Text('Add Upcoming Birthday', 
                  style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold, color: Colors.pink[800])),
                SizedBox(height: 20),
                inputField('Name', Icons.person, nameCtrl),
                SizedBox(height: 15),
                inputField('Date (DD/MM/YYYY)', Icons.cake, dateCtrl),
                SizedBox(height: 20),
                ElevatedButton(
                  onPressed: addBirthday,
                  child: Text('Add Birthday', style: TextStyle(fontSize: 18)),
                  style: ElevatedButton.styleFrom(
                    backgroundColor: Colors.pink,
                    foregroundColor: Colors.white,
                    padding: EdgeInsets.all(15),
                    minimumSize: Size(double.infinity, 50),
                  ),
                ),
                SizedBox(height: 30),
                Text('Upcoming Birthdays', 
                  style: TextStyle(fontSize: 22, fontWeight: FontWeight.bold, color: Colors.pink[800])),
                SizedBox(height: 15),
                Expanded(
                  child: birthdays.isEmpty
                    ? Center(child: Text('No birthdays added yet!', 
                        style: TextStyle(fontSize: 16, color: Colors.grey[600])))
                    : ListView.builder(
                        itemCount: birthdays.length,
                        itemBuilder: (context, index) {
                          return Card(
                            margin: EdgeInsets.symmetric(vertical: 8),
                            child: ListTile(
                              leading: CircleAvatar(
                                backgroundColor: Colors.pink,
                                child: Icon(Icons.cake, color: Colors.white),
                              ),
                              title: Text(birthdays[index]['name']!, 
                                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold)),
                              subtitle: Text(birthdays[index]['date']!, 
                                style: TextStyle(fontSize: 16)),
                              trailing: Row(
                                mainAxisSize: MainAxisSize.min,
                                children: [
                                  IconButton(
                                    icon: Icon(Icons.celebration, color: Colors.orange),
                                    onPressed: () => showBirthdayWish(birthdays[index]['name']!),
                                  ),
                                  IconButton(
                                    icon: Icon(Icons.delete, color: Colors.red),
                                    onPressed: () => deleteBirthday(index),
                                  ),
                                ],
                              ),
                            ),
                          );
                        },
                      ),
                ),
              ],
            ),
          ),
          
          if (showWish)
            Center(
              child: ScaleTransition(
                scale: scaleAnim,
                child: Container(
                  padding: EdgeInsets.all(30),
                  margin: EdgeInsets.all(30),
                  decoration: BoxDecoration(
                    gradient: LinearGradient(
                      colors: [Colors.pink, Colors.purple, Colors.orange],
                    ),
                    borderRadius: BorderRadius.circular(20),
                  ),
                  child: Column(
                    mainAxisSize: MainAxisSize.min,
                    children: [
                      Icon(Icons.cake, size: 70, color: Colors.white),
                      SizedBox(height: 15),
                      Text('ðŸŽ‰ Happy Birthday! ðŸŽ‰', 
                        style: TextStyle(fontSize: 26, fontWeight: FontWeight.bold, color: Colors.white)),
                      SizedBox(height: 10),
                      Text(wishName, 
                        style: TextStyle(fontSize: 30, fontWeight: FontWeight.bold, color: Colors.yellow)),
                      SizedBox(height: 10),
                      Text('ðŸŽ‚ Wishing you a wonderful day! ðŸŽ‚', 
                        style: TextStyle(fontSize: 16, color: Colors.white)),
                    ],
                  ),
                ),
              ),
            ),
        ],
      ),
    );
  }
}
